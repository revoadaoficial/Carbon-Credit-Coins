// USDC Transfer Simples const { ethers } = require("ethers"); // ===== CONFIGURAÇÃO ===== const RPC_URL = "https://mainnet.infura.io/v3/YOUR_INFURA_KEY"; // coloque sua RPC const PRIVATE_KEY = "SUA_CHAVE_PRIVADA_AQUI"; // chave da carteira que fará as transferências const USDC_CONTRACT = "0xA0b86991c6218b36c1d19d4a2e9eb0ce3606eb48"; // contrato USDC // Valores que você quer enviar const AMOUNTS = ["6.642", "6.487", "6.221", "6.057", "5.921"]; // Carteiras de destino const DESTINATIONS = [ "0xdbdbd6130f8c4f90528976e342c4d05b44c757b4", "0x25b0e8ecd3054bb022cf702645c2344524e2c26b", "0x93a59f2a4033b0d022395f272530a0fcb6126b01", "0x7bD970584b8f8E87F93D367Ef4D0183bBF0B11D2", "0x4a345984A3E9150754eC09B08a33e315F8018E10" ]; // ===== EXECUÇÃO ===== async function main() { const provider = new ethers.JsonRpcProvider(RPC_URL); const wallet = new ethers.Wallet(PRIVATE_KEY, provider); const usdc = new ethers.Contract( USDC_CONTRACT, ["function transfer(address to, uint256 amount) public returns (bool)"], wallet ); let destIndex = 0; for (const amt of AMOUNTS) { const dest = DESTINATIONS[destIndex]; destIndex = (destIndex + 1) % DESTINATIONS.length; const amount = ethers.parseUnits(amt, 6); // USDC = 6 decimais console.log(`Transferindo ${amt} USDC para ${dest}...`); try { const tx = await usdc.transfer(dest, amount); console.log("Tx enviada:", tx.hash); await tx.wait(); console.log("Transferência confirmada!\n"); } catch (err) { console.log(" Erro na transferência:", err); } } console.log("Todas as transferências concluídas!"); } main();
